\begin{problem}{sample-limit}[topic=limits, difficulty=easy]
	Evaluate $\displaystyle\lim_{x\to 2}(3x+1)$.
	\begin{solution}
	Direct substitution: $3(2)+1 = 7$.
	\end{solution}
\end{problem}

\begin{problem}{sample-deriv}[topic=derivatives, difficulty=medium]
	Differentiate $f(x) = x^3 - 2x$.
	\begin{solution}
	$f'(x) = 3x^2 - 2$.
	\end{solution}
\end{problem}
