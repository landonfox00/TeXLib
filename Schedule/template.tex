% =============================================================================
% schedule template — semester schedule grid
% =============================================================================
%
% Compile with:
%   lualatex template.tex
%
% Requires coursemeta.tex at an ancestor directory providing:
%   institution, course-subject, course-number, course-title, course-section,
%   season, year, start-date, end-date, final-date, final-time, lecture-days.

\documentclass{schedule}

\meta{
	landscape    = true,
	lecture-caps = {1.0, 1.0},   % equal weight for each lecture-days bucket
	quiz-days    = T,            % weekly Tuesday quiz
	% month-pages = true,        % one table per month, each on its own page
}

\begin{document}
\maketitle
\begin{schedule}

	% --- Holidays -----------------------------------------------------
	% Fall 2026 dates, matching the Fall start/end metadata in
	% examples/Math181-Fall2026/coursemeta.tex (and the smoke-test stub).
	\holiday{9-7}{Labor Day}
	\holiday{11-25}[11-27]{Thanksgiving Break}
	\holiday{12-11}{Prep Day}

	% --- Quiz overrides ----------------------------------------------
	\noquiz{8-25}
	\quiz[date=8-26]

	% --- Unit 1 -------------------------------------------------------
	\syllabus
	\section{R.1}
	\section{R.2}
	\section{R.3}
	\section[2]{R.4}
	\examreview
	\exam[noquiz]

	% --- Unit 2 -------------------------------------------------------
	\section{2.1}
	\section[2]{2.2}
	\examreview
	\exam

	% --- Final week ---------------------------------------------------
	\finalreview
	\finalsweek[12-14][5]
	\winterbreak

\end{schedule}
\end{document}
