\begin{problem}{deriv-mc}
	What is $\dfrac{d}{dx}\!\left[x^2\right]$?
	\begin{choices}
		\cchoice $2x$
		\choice  $x$
		\choice  $\dfrac{x^3}{3}$
		\choice  $2$
	\end{choices}
	\begin{solution} By the power rule, $2x$. \end{solution}
\end{problem}

\begin{problem}{int-fr}
	Evaluate $\displaystyle\int_0^2 3x^2 \, dx$.
	\begin{solution} $\bigl[x^3\bigr]_0^2 = 8$. \end{solution}
\end{problem}
