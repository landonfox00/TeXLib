\begin{problem}{int-fr}
	Evaluate $\displaystyle\int_0^1 x \, dx$.
	\begin{solution} By the power rule, $\bigl[\tfrac{x^2}{2}\bigr]_0^1 = \tfrac{1}{2}$. \end{solution}
\end{problem}

\begin{problem}{deriv-mc}
	Which is $\dfrac{d}{dx}\!\left[\sin x\right]$?
	\begin{choices}
		\cchoice $\cos x$
		\choice  $-\cos x$
		\choice  $-\sin x$
	\end{choices}
	\begin{solution} The derivative of $\sin x$ is $\cos x$. \end{solution}
\end{problem}
