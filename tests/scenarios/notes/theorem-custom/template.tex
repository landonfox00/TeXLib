\section{Introduction}
These boxes use the customised look: no background tint, a teal theorem rule and
a purple definition rule.

\begin{definition}[Continuity]
	A function $f$ is continuous at $a$ if $\lim_{x \to a} f(x) = f(a)$.
\end{definition}

\begin{theorem}[Intermediate Value]
	If $f$ is continuous on $[a,b]$ and $N$ lies between $f(a)$ and $f(b)$,
	then there exists $c \in (a,b)$ with $f(c) = N$.
\end{theorem}
