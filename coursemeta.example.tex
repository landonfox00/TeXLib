% coursemeta.example.tex
%
% Example course-metadata file. Copy to `coursemeta.tex` in your course
% directory (or any ancestor up to three levels above the document) and
% course-metadata.sty will auto-load it. Every TeXLib document class picks
% the values up via \GetCourse, \GetInstructor, \GetTerm, etc.
%
% All keys are optional — define only what your documents reference. Any
% key you set becomes accessible as \Get<TitleCasedKey> (e.g. start-date
% becomes \GetStartDate). The schedule class is the strictest consumer:
% it requires lecture-days, start-date, end-date.

\metasetup{
	institution     = {University of Nevada, Reno},
	instructor      = {Your Name},
	season          = Fall,
	year            = 2026,
	course-subject  = Math,
	course-number   = 181,
	course-title    = {Calculus I},
	course-section  = 1001,
	course-room     = {DMSC 100},
	lecture-days    = MWF,
	lecture-times   = {9:00-9:50am},
	start-date      = 8-24,
	end-date        = 12-8,
	final-date      = 12-15,
	final-time      = {9:45-11:45am},
	% bank-path is resolved relative to this file's directory — both
	% autoexam.cls and quiz.cls prepend \GetCourseMetaDir so the same
	% string works at any document depth.
	bank-path       = {Bank/bank.tex},
	% gradebook-path lets bare \gradebook (report-card.cls) find the course
	% gradebook. Resolved like bank-path; an .xlsx value is mapped to the
	% .csv the TeXLib builder exports from it.
	gradebook-path  = {gradebook.xlsx},
	% --- Optional course-wide defaults (a document's class options override) ----
	% Instructor contact + scheduling fields. These were historically options of
	% the syllabus class; set here, they become defaults for every document.
	% instructor-email = {you@unr.edu},      % alias: email
	% office           = {NLLC 146F},
	% office-hours     = {MWF 9--10am},
	% class-time       = {MWF 9:00--9:50am}, % syllabus "Time" row; omit to derive
	%                                        % it from lecture-days + lecture-times
	% short-title      = {Math 181 Calc I},  % short header title for the syllabus
	% Per-exam dates feed both autoexam (matched by exam-number) and the syllabus
	% \examdatetable; a Final exam (exam-number = Final) inherits final-date above.
	% exam1-date       = {Sep 19, 2026},
	% exam2-date       = {Oct 17, 2026},
	% exam3-date       = {Nov 14, 2026},
}
