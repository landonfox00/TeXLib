\section{Introduction}

This is a sample of the boxed environments and the magic
\texttt{solution} box.

\begin{definition}[Continuity]
	A function $f$ is continuous at $a$ if
	$\lim_{x \to a} f(x) = f(a)$.
\end{definition}

\begin{theorem}[Intermediate Value]
	If $f$ is continuous on $[a,b]$ and $N$ lies between $f(a)$
	and $f(b)$, then there exists $c \in (a,b)$ with $f(c) = N$.
\end{theorem}

\begin{exercise}
	Show that the equation $x^3 - x - 1 = 0$ has a solution in
	$(1, 2)$.
\end{exercise}

\begin{solution}[2.5in]
	Let $f(x) = x^3 - x - 1$. Then $f(1) = -1 < 0$ and
	$f(2) = 5 > 0$. Since $f$ is continuous, the IVT gives a
	root in $(1, 2)$.
\end{solution}

\begin{challenge}
	Generalize: when does $x^n - x - 1$ have a real root?
\end{challenge}
