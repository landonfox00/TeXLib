% Exam 01 — Limits, an example randomized multi-version exam.
%
% Course metadata (title, course number, section, term) is picked up from
% coursemeta.tex in this directory; only the per-exam fields are set below.
%
% Build the collated A/B/C student exams:  lualatex exam-01.tex
% Build one version's student copy:         lualatex "\def\Version{A}\input{exam-01.tex}"
% Build the answer keys (A/B/C, solutions): lualatex "\def\ShowKey{}\input{exam-01.tex}"
% Build one version's answer key:           lualatex "\def\Version{A}\def\ShowKey{}\input{exam-01.tex}"

% exam-date is intentionally NOT set here: with exam-number = 1, the cover date
% comes from coursemeta's exam1-date — the same source the syllabus
% \examdatetable reads — so a reschedule is one edit. A local exam-date option
% (class option or \meta) would override it.
\documentclass[exam-number = 1]{autoexam}

% Three labeled versions. \shuffle reorders the questions (and each MC problem's
% options) independently per version (seeded by the version letter), so one
% authored order below yields A/B/C in different sequence — no per-version files.
\versions{A, B, C}
\shuffle

\loadbank{bank.tex}

\begin{document}
\maketitle

\begin{problems}
	\problem{topic=limit}
	\problem{topic=continuity}
	\problem{topic=ivt}
\end{problems}

\end{document}
