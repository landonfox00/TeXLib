\documentclass[quiz-number = 1]{quiz}

\begin{problem}{limit-poly}[topic=limits, diff=easy]
	Evaluate $\displaystyle \lim_{x \to 2} (3x^2 - x + 4)$.
	\begin{solution} $14$
	\end{solution}
\end{problem}

\begin{problem}{linear-eq}[topic=algebra, diff=easy]
	Solve $2x + 5 = 11$ for $x$.
	\begin{solution} $x = 3$
	\end{solution}
\end{problem}

\begin{document}

\begin{questions}

\question \getproblem{limit-poly}
	\vspace{\stretch{1}}

\question \getproblem{linear-eq}
	\vspace{\stretch{1}}
