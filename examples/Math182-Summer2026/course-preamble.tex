% --- Notation shortcuts -----------------------------------------------------
\newcommand{\ds}{\displaystyle}

% Differentials: a thin space then the italic "d" of the variable.
\newcommand{\dx}{\,dx}
\newcommand{\dtheta}{\,d\theta}

% \eval{expr}{lower}{upper} — an evaluation bar  [expr]_{lower}^{upper}.
\newcommand{\eval}[3]{\left.#1\right|_{#2}^{#3}}

% Compact integral:  \int[<lower>][<upper>]{<integrand>}[<var>]
%   - both bounds optional (omit for an indefinite integral),
%   - integrand mandatory,
%   - differential variable optional (defaults to x).
% Examples:
%   \int{f(x)}              -> \int f(x)\,dx
%   \int{g(\theta)}[\theta] -> \int g(\theta)\,d\theta
%   \int[0][2]{h(x)}        -> \int_{0}^{2} h(x)\,dx
\let\txlibintop\int
\RenewDocumentCommand{\int}{o o m o}{%
	\txlibintop
		\IfValueT{#1}{_{#1}}%
		\IfValueT{#2}{^{#2}}%
	#3\,d\IfValueTF{#4}{#4}{x}%
}
